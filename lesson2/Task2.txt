#! /bin/bash
# Программа производит обмен значений соседних элементов списка
l = input('Введите числа для списка через пробел: ').split()
l2 = []
for i in range(1, len(l), 2): # создаем новый список со значениями для замены
    l2.append(l[i])
if len(l) % 2 == 0: # если количество элементов в изначальном списке четное просто выводим новый список
    print(l2)
else:
    l2.append(l[-1]) # если количество элементов в изначальном списке нечетное добавляем в конец последний элемент списка l
    print(l2)
